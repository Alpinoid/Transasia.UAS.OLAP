SELECT
	{
		[Measures].[Количество]
		,[Measures].[Количество коробок]
		,[Measures].[Сумма продаж]
	} ON COLUMNS
	,
	NON EMPTY
		(
		[13 Документы Реализация товаров].[ID].[ID]
		--,[13 Документы Реализация товаров].[Дата документа].[Дата документа]
		--,[13 Документы Реализация товаров].[Номер документа].[Номер документа]
		,[05 Контрагенты].[ID].[ID]
		,[06 Торговые точки].[ID].[ID]
		,[07 Маршруты].[ID].[ID]
		--,[13 Документы Реализация товаров].[Дата доставки документа].[Дата доставки документа]
		--,[13 Документы Реализация товаров].[Дата оплаты документа].[Дата оплаты документа]
		)
	ON ROWS
FROM [Продажи]
WHERE (
	[07-01-01 Системы мобильных продаж].[Системы мобильных продаж].[Система мобильных продаж].&[Yukon-PG]
	,[13 Документы Реализация товаров].[Бонусный документ].[Бонусный документ].&[0]
	,StrToMember('[13 Документы Реализация товаров].[Дата документа].[Дата документа].&['+Format(DateAdd('M', -3, NOW()), 'yyyy-MM-dd') + 'T00:00:00]').Parent.FirstChild
	:StrToMember('[13 Документы Реализация товаров].[Дата документа].[Дата документа].&['+Format(DateAdd('D', 3, NOW()), 'yyyy-MM-dd') + 'T00:00:00]')
)
